<script>
// ─── UI ────────────────────────────────────────────────────────────────────
const video    = document.getElementById('remote-video');
const waiting  = document.getElementById('waiting');
const modeBtn  = document.getElementById('mode-btn');

const sFps  = document.getElementById('s-fps');
const sRtt  = document.getElementById('s-rtt');
const sJit  = document.getElementById('s-jit');
const sLost = document.getElementById('s-lost');
const sBuf  = document.getElementById('s-buf');

let currentMode = 0;
const modes = [
    { cls: '',                  txt: 'Mod: Orijinal' },
    { cls: 'mode-center-crop',  txt: 'Mod: Center Crop' },
    { cls: 'mode-fit',          txt: 'Mod: Fit' },
];
function toggleMode() {
    video.classList.remove(modes[currentMode].cls);
    currentMode = (currentMode + 1) % modes.length;
    video.classList.add(modes[currentMode].cls);
    modeBtn.textContent = modes[currentMode].txt;
}

let hudVisible = true;
function toggleHud() {
    hudVisible = !hudVisible;
    document.getElementById('hud').style.opacity = hudVisible ? '1' : '0';
}

// ─── PeerJS ────────────────────────────────────────────────────────────────
const peer = new Peer({
    config: { iceServers: [{ urls: 'stun:stun.l.google.com:19302' }] }
});

peer.on('open', id => {
    document.getElementById('my-id').textContent = id;
});

peer.on('call', call => {
    call.answer();

    call.on('stream', remoteStream => {
        // ── 1. Stream'i hemen bağla, play() tetikle ──
        video.srcObject = remoteStream;

        // ── 2. Jitter buffer sıfırla (en kritik alıcı optimizasyonu) ──
        applyReceiverOptimizations(call);

        // ── 3. Video hazır olur olmaz oluşabilecek en erken play() ──
        const tryPlay = () => {
            video.play().catch(() => {});
        };
        // readyState >= 1 (HAVE_METADATA) → yeterli, beklemeden başlat
        if (video.readyState >= 1) {
            tryPlay();
        } else {
            video.addEventListener('loadedmetadata', tryPlay, { once: true });
        }

        // ── 4. Bekleme ekranını kaldır ──
        waiting.classList.add('hidden');

        // ── 5. İstatistikleri başlat ──
        startStats(call);
        startFrameStats();
    });
});

// ─── Alıcı Optimizasyonları ────────────────────────────────────────────────
function applyReceiverOptimizations(call) {
    // PeerJS 1.5.x iç RTCPeerConnection referansı
    const pc = call.peerConnection || call._pc;
    if (!pc) return;

    // Receiver'lar hemen hazır olmayabilir, polling ile zorla
    function forceZeroBuffer() {
        const receivers = pc.getReceivers();
        let applied = false;
        receivers.forEach(receiver => {
            if (!receiver.track || receiver.track.kind !== 'video') return;
            applied = true;

            // jitterBufferTarget: Chromium M112+ — 0 = buffer yok, anında ilet
            if ('jitterBufferTarget' in receiver) {
                receiver.jitterBufferTarget = 0;
            }
            // jitterBufferDelayHint: standart API
            if ('jitterBufferDelayHint' in receiver) {
                receiver.jitterBufferDelayHint = 0;
            }
        });
        return applied;
    }

    // İlk deneme
    if (!forceZeroBuffer()) {
        // Track henüz gelmemişse kısa aralıklarla tekrar dene
        let attempts = 0;
        const retry = setInterval(() => {
            if (forceZeroBuffer() || ++attempts > 20) clearInterval(retry);
        }, 100);
    }

    // Bağlantı kararlı olduktan sonra bir kez daha uygula (race condition önlemi)
    pc.addEventListener('connectionstatechange', () => {
        if (pc.connectionState === 'connected') forceZeroBuffer();
    });

    // ── Video element latency ipuçları ──
    video.playsInline = true;

    // Chromium: HTMLMediaElement.latencyHint (non-standard, varsa kullan)
    if ('latencyHint' in video) video.latencyHint = 'interactive';

    // preservesPitch: false → pitch düzeltmesi için extra buffer tutulmasın
    if ('preservesPitch' in video) video.preservesPitch = false;

    // disableRemotePlayback → cast/airplay overhead'ini kaldır (HTML attribute ile zaten set edildi)
    video.disableRemotePlayback = true;
}

// ─── Frame Zamanlama (requestVideoFrameCallback) ───────────────────────────
let lastPresentTime = 0;
let frameCount = 0;
let fpsTimer = performance.now();

function startFrameStats() {
    if (!('requestVideoFrameCallback' in video)) {
        // Fallback: rAF ile FPS tahmin et
        const rAFfps = () => {
            frameCount++;
            const now = performance.now();
            if (now - fpsTimer >= 500) {
                sFps.textContent = Math.round(frameCount / (now - fpsTimer) * 1000);
                frameCount = 0;
                fpsTimer = now;
            }
            requestAnimationFrame(rAFfps);
        };
        requestAnimationFrame(rAFfps);
        return;
    }

    // requestVideoFrameCallback: gerçek decode→display süresi
    // metadata.expectedDisplayTime - metadata.mediaTime farkı gerçek pipeline gecikmesini verir
    const onFrame = (now, metadata) => {
        frameCount++;

        // Gerçek frame işleme süresi
        if (lastPresentTime > 0) {
            const delta = metadata.presentationTime - lastPresentTime;
            // delta ~ 16.6ms ise 60fps, iyi
        }
        lastPresentTime = metadata.presentationTime;

        // FPS hesabı
        const elapsed = now - fpsTimer;
        if (elapsed >= 500) {
            const fps = Math.round(frameCount / elapsed * 1000);
            sFps.textContent = fps;
            sFps.className = 'stat-val' + (fps < 30 ? ' bad' : fps < 50 ? ' warn' : '');
            frameCount = 0;
            fpsTimer = now;
        }

        video.requestVideoFrameCallback(onFrame);
    };
    video.requestVideoFrameCallback(onFrame);
}

// ─── WebRTC Stats ──────────────────────────────────────────────────────────
function startStats(call) {
    const pc = call.peerConnection || call._pc;
    if (!pc) return;

    setInterval(async () => {
        try {
            const stats = await pc.getStats();
            stats.forEach(r => {
                // RTT
                if (r.type === 'candidate-pair' && r.state === 'succeeded') {
                    const rtt = r.currentRoundTripTime;
                    if (rtt != null) {
                        const ms = (rtt * 1000).toFixed(1);
                        sRtt.textContent = ms;
                        sRtt.className = 'stat-val' + (ms > 20 ? ' bad' : ms > 10 ? ' warn' : '');
                    }
                }
                // Jitter, kayıp paket, jitter buffer
                if (r.type === 'inbound-rtp' && r.kind === 'video') {
                    // Jitter
                    if (r.jitter != null) {
                        const jMs = (r.jitter * 1000).toFixed(1);
                        sJit.textContent = jMs;
                        sJit.className = 'stat-val' + (jMs > 5 ? ' bad' : jMs > 2 ? ' warn' : '');
                    }
                    // Kayıp paket
                    if (r.packetsLost != null) {
                        sLost.textContent = r.packetsLost;
                        sLost.className = 'stat-val' + (r.packetsLost > 10 ? ' bad' : r.packetsLost > 0 ? ' warn' : '');
                    }
                    // Jitter buffer gecikme tahmini
                    if (r.jitterBufferDelay != null && r.jitterBufferEmittedCount != null && r.jitterBufferEmittedCount > 0) {
                        const bufMs = ((r.jitterBufferDelay / r.jitterBufferEmittedCount) * 1000).toFixed(1);
                        sBuf.textContent = bufMs;
                        sBuf.className = 'stat-val' + (bufMs > 10 ? ' bad' : bufMs > 5 ? ' warn' : '');
                    }
                }
            });
        } catch (_) {}
    }, 1000);
}
